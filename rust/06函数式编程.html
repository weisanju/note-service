#![allow(unused)]
fn main() {
// 函数式的写法
let sum_of_squared_odd_numbers: u32 =
    (0..).map(|n| n * n)             // 所有自然数取平方
         .take_while(|&n| n < upper) // 取小于上限的
         .filter(|&n| is_odd(n))     // 取奇数
         .fold(0, |sum, i| sum + i); // 最后加起来
println!("functional style: {}", sum_of_squared_odd_numbers);
}
